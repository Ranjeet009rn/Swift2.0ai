### **Core Tables Structure**

#### **Users Table**

```sql
CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id VARCHAR(50) UNIQUE NOT NULL,
    name VARCHAR(255) NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    mobile VARCHAR(15),
    sponsor_id VARCHAR(50),
    placement_id VARCHAR(50),
    placement_position ENUM('left', 'right'),
    selected_package VARCHAR(50),
    role ENUM('user', 'franchise', 'admin') DEFAULT 'user',
    status ENUM('active', 'inactive', 'blocked') DEFAULT 'inactive',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    INDEX idx_user_id (user_id),
    INDEX idx_sponsor (sponsor_id),
    INDEX idx_placement (placement_id),
    INDEX idx_role (role),
    INDEX idx_status (status)
);
```

#### **Franchise Users Table**

```sql
CREATE TABLE franchise_users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id VARCHAR(50) UNIQUE NOT NULL,
    franchisee_name VARCHAR(255) NOT NULL,
    contact_person_name VARCHAR(255) NOT NULL,
    mobile_number VARCHAR(15) NOT NULL,
    franchise_type ENUM('district', 'state', 'national') NOT NULL,
    area_territory VARCHAR(255),
    business_address TEXT,
    city VARCHAR(100),
    state VARCHAR(100),
    pincode VARCHAR(10),
    pan_number VARCHAR(20),
    govt_id_type VARCHAR(50),
    govt_id_number VARCHAR(50),
    id_proof_path VARCHAR(500),
    bank_name VARCHAR(255),
    account_holder_name VARCHAR(255),
    account_number VARCHAR(50),
    ifsc_code VARCHAR(20),
    application_status ENUM('pending', 'approved', 'rejected') DEFAULT 'pending',
    admin_remarks TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    approved_at TIMESTAMP NULL,
    
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE,
    INDEX idx_status (application_status),
    INDEX idx_user (user_id)
);
```

#### **Hierarchical Users Table**

```sql
CREATE TABLE hierarchical_users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id VARCHAR(50) UNIQUE NOT NULL,
    parent_id VARCHAR(50),
    position ENUM('left', 'right'),
    level INT DEFAULT 1,
    left_count INT DEFAULT 0,
    right_count INT DEFAULT 0,
    left_business DECIMAL(15,2) DEFAULT 0,
    right_business DECIMAL(15,2) DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE,
    INDEX idx_parent (parent_id),
    INDEX idx_level (level)
);
```

#### **E-Pins Table**

```sql
CREATE TABLE epins (
    id INT AUTO_INCREMENT PRIMARY KEY,
    pin VARCHAR(20) UNIQUE NOT NULL,
    package VARCHAR(50) NOT NULL,
    amount DECIMAL(10,2) NOT NULL,
    status ENUM('unused', 'used') DEFAULT 'unused',
    owner_id VARCHAR(50),
    allocated_to ENUM('admin', 'franchise') DEFAULT 'admin',
    used_by VARCHAR(50),
    used_date TIMESTAMP NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    INDEX idx_pin (pin),
    INDEX idx_status (status),
    INDEX idx_owner (owner_id)
);
```
